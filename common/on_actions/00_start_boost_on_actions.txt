on_game_started = {
  on_actions = { start_boost }
}

on_yearly_pulse = {
  on_actions = { start_boost2 }
}

on_game_started = {
  on_actions = { start_boost3 }
}

on_game_started = {
  on_actions = { holy_roman_emperor }
}

on_game_started = {
  on_actions = { remove_primary_cultures }
}

start_boost = {
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier
        days = 365
      }
    }
  }
}

start_boost2 = {
  trigger = { game_date <= 1544.1.1 }
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier2
        days = 365
      }
    }
  }
}

start_boost3 = {
  effect = {
    every_country = {
      every_scope_state = {
        limit = { 
          is_coastal = yes 
          owner = { has_technology_researched = admiralty }
        }
        create_building = {
          building = "building_naval_base"
          level = 1
          reserves = 1
        }
      }
      capital = {
        create_building = {
          building = "building_construction_sector"
          level = 1
          reserves = 1
        }
        create_building = {
          building = "building_barracks"
          level = 1
          reserves = 1
        }
      }
    }
  }
}

holy_roman_emperor = {
  effect = {
    c:AUS = {
      add_modifier = {
        name = holy_roman_emperor
        days = 17520
        is_decaying = yes
      }
    }
  }
}

remove_primary_cultures = {
  effect = {
    c:X22 = {

    }
    c:FRA = {
      remove_primary_culture = cu:wallonian

    c:ENG = {
      remove_primary_culture = cu:french
    }
    c:LIT = {
      remove_primary_culture = cu:ukrainian
      remove_primary_culture = cu:russian
    }
    c:POL = {
      remove_primary_culture = cu:russian
      remove_primary_culture = cu:ukrainian
    }
    c:M34 = {
      remove_primary_culture = cu:ukrainian
    }
    c:HUN = {
      remove_primary_culture = cu:romanian

    }
  }
}
