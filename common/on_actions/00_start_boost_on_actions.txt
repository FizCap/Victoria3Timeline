on_game_started = {
  on_actions = { start_boost }
}

on_game_started = {
  on_actions = { start_boost2 }
}

on_game_started = {
  on_actions = { start_boost3 }
}

on_game_started = {
  on_actions = { holy_roman_emperor }
}

start_boost = {
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier
        days = 365
      }
    }
  }
}

start_boost2 = {
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier2
        days = 1825
      }
    }
  }
}

start_boost3 = {
  effect = {
    every_country = {
      capital = {
        create_building = {
          building = "building_construction_sector"
          level = 1
          reserves = 1
        }
      }
    }
  }
}

holy_roman_emperor = {
  effect = {
    c:AUS = {
      add_modifier = {
        name = holy_roman_emperor
        days = 17520
      }
    }
  }
}
