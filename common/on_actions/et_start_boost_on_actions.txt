on_game_started = {
  on_actions = { start_boost }
}

on_yearly_pulse = {
  on_actions = { start_boost2 }
}

on_game_started = {
  on_actions = { start_boost3 }
}

on_yearly_pulse = { 
  on_actions = { start_boost4 } 
}

on_game_started = {
  on_actions = { remove_primary_cultures }
}



start_boost = {
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier
        days = 365
      }
    }
  }
}

start_boost2 = {
  trigger = { game_date <= 1544.1.1 }
  effect = {
    every_country = {
      add_modifier = {
        name = start_boost_country_modifier2
        days = 365
      }
    }
  }
}

start_boost3 = {
  effect = {
    every_country = {
      every_scope_state = {
        limit = { 
          is_coastal = yes 
          owner = { has_technology_researched = admiralty }
        }
        create_building = {
          building = "building_naval_base"
          level = 1
          reserves = 1
        }
      }
      capital = {
        create_building = {
          building = "building_construction_sector"
          level = 1
          reserves = 1
        }
        create_building = {
          building = "building_barracks"
          level = 1
          reserves = 1
        }
        create_building = {
          building = "building_trade_center"
          level = 1
          reserves = 1
        }
      }
    }
  }
}

start_boost4 = {
  effect = {
    every_country = {
      limit = { total_population < 50000 }
      add_modifier = {
        name = start_boost_country_modifier4
        days = 365
      }
    }
    every_decentralized_country = {
      limit = { total_population < 5000 }
      add_modifier = {
        name = start_boost_country_modifier4
        days = 365
      }
    }
  }
}
